// Function to add zoom effect to pricing boxes function zoomBox(box) { // First remove zoom class from all boxes const allBoxes = document.querySelectorAll('.pricing-box'); allBoxes.forEach(b => b.classList.remove('zoom')); // Add zoom class to the clicked box box.classList.add('zoom'); // If on mobile, scroll to the zoomed box if (window.innerWidth <= 576) { box.scrollIntoView({behavior: 'smooth', block: 'center'}); } } // Apply hover effects on desktop only function applyHoverEffects() { if (window.innerWidth > 768) { const pricingBoxes = document.querySelectorAll('.pricing-box'); pricingBoxes.forEach(box => { box.addEventListener('mouseover', function() { this.style.transform = 'scale(1.05)'; this.style.boxShadow = '0 10px 30px rgba(255, 255, 255, 0.3)'; }); box.addEventListener('mouseout', function() { if (!this.classList.contains('zoom')) { this.style.transform = ''; this.style.boxShadow = ''; } }); }); } } // Run when the DOM is fully loaded document.addEventListener('DOMContentLoaded', function() { applyHoverEffects(); // Reset zoom when clicking elsewhere on the page document.addEventListener('click', function(event) { if (!event.target.closest('.pricing-box')) { const allBoxes = document.querySelectorAll('.pricing-box'); allBoxes.forEach(box => { box.classList.remove('zoom'); box.style.transform = ''; box.style.boxShadow = ''; }); } }); // Reapply effects on window resize window.addEventListener('resize', applyHoverEffects); });
